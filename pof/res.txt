                    function LoadFeedItems(json)
                    {
                        var $items = jQuery('#meetme-items');
                        var currentCount = $items.children().length;
                        var itemCount = json.length;

                        var content = "";

                        // if we don't have any entries, add a placeholder / empty data state
                        if (itemCount == 0 &amp;&amp; firstload)
                        {
                            content = emptyStateTemplate;
                        }
                        else
                        {
                            for (var i = 0; i &lt; itemCount; i++)
                            {
                                var bgClass = "meetme_unread";
                                if (json[i].IsRead)
                                    bgClass = "meetme_read";

                                // the backend only really produces MeetMeMutual, but for forwards compatibility
                                var notificationType;
                                switch (json[i].Type) {
                                    case 1:
                                        notificationType = "Notification_MeetYou";
                                        break;
                                    case 3:
                                        notificationType = "Notification_MeetMeMutual";
                                        break;
                                    default:
                                        notificationType = "Unknown";
                                }

                                // all of the other entries are mutual meet me links
                                content += "&lt;li id='meetme_" + (currentCount + i) + "' class='meetme_listentry " + bgClass + "' data-notification-type='" + notificationType + "'&gt;";
                                content += "&lt;a class='meetme_feed_clickable' href='pt_viewprofile.aspx?profile_id=" + json[i].ProfileID + "&amp;alertmutual=1'&gt;";
                                content += "&lt;span class='meetme_feed_image_wrapper'&gt;&lt;img class='meetme_feed_image' src='" + json[i].UserImage + "'&gt;&lt;/span&gt;";
                                content += "&lt;span class='meetme_feed_content_wrapper'&gt;";
                                content += "&lt;span class='meetme_feed_message'&gt;" + json[i].Message + "&lt;/span&gt;";
                                content += "&lt;span class='meetme_feed_time'&gt;" + json[i].TimeStamp + "&lt;/span&gt;";
                                content += "&lt;/span&gt;";
                                content += "&lt;span class='meetme_feed_icon'&gt;&lt;img src='" + json[i].Icon + "' \&gt;&lt;/span&gt;";
                                content += "&lt;/a&gt;";
                                content += "&lt;/li&gt;";
                            }
                            if (itemCount &gt; 0) {
                                // save the id of the last record we recieved so we can do infinite scroll for the next batch
                                lastid = json[itemCount - 1].MessageID;
                            }
                            else {
                                ready = false;      // no more results
                                //console.log("done infinite scroll, no more results");
                            }
                        }
                        return content;
                    }
